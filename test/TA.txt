    //: go
    package main
    import (
        "fmt"
        "runtime"
    )
    func main() {
        fmt.Print("Go runs on ")
        switch os := runtime.GOOS; os {
        case "darwin":
            fmt.Println("OS X.")
        case "linux":
            fmt.Println("Linux.")
        default:
            // freebsd, openbsd,
            // plan9, windows...
            fmt.Printf("%s.", os)
        }
    }

# 段落

go 中的字符串类型是真 jb 坑, 比如 `s := "go语言"`,
len(s) 的长度可不是 4, 它输出的是 s 的字节长度,
要想获得表面上的长度, 需要使用 `len(([]rune(s))) `, 这个时候才会输出 4.
